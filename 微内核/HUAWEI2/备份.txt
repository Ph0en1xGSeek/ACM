#include <stdio.h>
#include <iostream>
#include <string.h>
#include <stack>
#define M 600
#define INF 1<<30

using namespace std;

int arr[M][M];
int matrix[M][M];
int V[M] = {0};
int checkV[M];
int ss[M];
int top = 0;

bool djikstra(int s, int t, int num)
{
    int i;
    bool isVisted[M + 1] = {0};
    isVisted[s] = 1;
    int d[M];
    int from[M];
    int mini, tmp;
    for(i = 0; i < num; i++)
    {
        d[i] = arr[s][i];
        if(d[i] < INF)
            from[i] = s;
    }
    for(i = 0; i < num; i++)
    {
        tmp = M;
        mini = INF;
        for(int j = 0; j < num; j++)
        {
            if(!isVisted[j] && d[j] < mini)
            {
                tmp = j;
                mini = d[j];
            }
        }
        isVisted[tmp] = 1;
        if(tmp == t || tmp == M)
            break;

        for(int j = 0; j <num; j++)
        if(!isVisted[j] && d[j] > arr[tmp][j] + d[tmp])
	    {
            from[j] = tmp;
            d[j] = arr[tmp][j] + d[tmp];
	    }
    }
    if(tmp != t)
    {
        return 0;
    }
    if(tmp == M)
        return 0;
    while(tmp != s)
    {
        ss[top++] = matrix[from[tmp]][tmp];
        tmp = from[tmp];
    }
    return 1;
}

int main(int argc, char **argv)
{
	int a, b, c, d;
	int s, t, num = 0, numV = 0;
	char tmp;
	for(int i = 0; i  < M; i++)
	    for(int j = 0; j < M; j++)
            arr[i][j] = INF;
	while(scanf("%d,%d,%d,%d", &a, &b, &c, &d) != EOF)
	{
	    if(d < arr[b][c])
	    {
            matrix[b][c] = a;
            arr[b][c] = d;
	    }
	    num = max(num, b);
	    num = max(num, c);
	}
	num++;
	scanf("%d, %d,", &s, &t);
	while(scanf("%d%c", &d, &tmp) && tmp == '|')
	{
	    numV++;
	    V[d] = 1;
	}
	numV++;
	V[d] = 1;


	for(int i = 0; i < num; i++)
	    if(!V[i])
        {
            if(i != t)
            {

                int cnt = 0;
                for(int j = 0; j < num; j++)
                    if(V[j] && arr[i][j] < INF)
                        cnt++;
                for(int j = 0; j < num; j++)
                    if(arr[j][i] < INF)
                    {
                        arr[j][i] -= 20;
                        arr[j][i] *= cnt;
                    }
            }
        }
        else
        {
            for(int  j = 0; j < num; j++)
            {
                if(arr[j][i] < INF)
                {
                    arr[j][i] -= 20;
                    arr[j][i] <<= 9;
                }
            }
        }

    for(int  j = 0; j < num; j++)
    {
        if(arr[j][t] < INF)
        {
            arr[j][t] += 10;
            arr[j][t] <<= 7;
        }
        if(j == s && arr[s][t] < INF)
            arr[j][t] <<= 6;
    }



	if(djikstra(s, t, num))
    {
        printf("%d", ss[--top]);
        while(top > 0)
        {
            printf("|%d", ss[--top]);
        }
        printf("\n");
    }
	else
    {
        printf("NA\n");
    }
	return 0;
}
